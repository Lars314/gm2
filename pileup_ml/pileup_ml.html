def getRandomSpline(minPulses = 0, maxPulses = 4,  # additional pulses in the island
                    deltaTmin = 0, deltaTmax = 25, 
                    minEscale = 1, maxEscale=10,
                    gainSag=None, 
                    verbosity=0, 
                    gaussianNoise=True, gaussianNoiseLevel = .001,
                    spline = spline, times = times, samplingrate = samplingrate,
                    ntailsamples = 150, randomizeTailLength = True,
                    normalize=True):
    '''
        Function to generate a random 'island' with a configurable number of pulses
    '''
    npulses = np.random.randint(minPulses, maxPulses+1)
    if(verbosity > 0):
        print( npulses, "pulses in this island")
        
    #generate energy scaling
    energyscalefactors = np.random.uniform(minEscale, maxEscale, size=npulses)
    
    if(randomizeTailLength):
        ntailsamples = ntailsamples + np.random.randint(-50,100)
        if(verbosity > 0):
            print(ntailsamples, "samples in this island")
        
    thisisland = np.append(np.zeros_like(spline), np.zeros(ntailsamples))
    thesetimes = np.append(times, np.array([times[times.size-1]+ samplingrate*i for i in range(1,ntailsamples+1)]))
    
    #add gaussian noise to each sample
    if(gaussianNoise):
        thisisland += np.random.normal(0,gaussianNoiseLevel, size=thisisland.size)
        
    if(npulses < 1):
        return thisisland, thesetimes, npulses, [], []
        
    timeoffsets = []
    for pi in range(npulses):
        deltat = np.random.uniform(deltaTmin, deltaTmax)
        timeoffsets.append(deltat)
        if(verbosity > 0):
            print("    Time offset from first pulse:", deltat)
    
        #use numpy interpolation to get the values in between our samples
        #get the int sample offset
        sample_offset = int(np.floor(deltat))
#         print(len(spline), len(times))
#         print(times)
        splinei = np.interp(times+(deltat)-sample_offset,times,spline)*energyscalefactors[pi]
#         print(splinei)
        thisisland[sample_offset:sample_offset+len(splinei)] += splinei
    
    if(normalize):
        thisisland /= np.sum(thisisland)
        
    return(thisisland, thesetimes, npulses, energyscalefactors, timeoffsets)
